# copy & paste at /etc/nginx/sites-available/default

# default 
server {
	listen 80 default_server;
	listen [::]:80 default_server;

	root /var/www/html;

	# Add index.php to the list if you are using PHP
	index index.html index.htm index.nginx-debian.html;

	server_name _;

	location / {
		# First attempt to serve request as file, then
		# as directory, then fall back to displaying a 404.
		try_files $uri $uri/ =404;
	}

}

# LEO CDP admin
upstream backend_admin_cdp {
  server 127.0.0.1:9070;
  #server 127.0.0.1:9071;
}
upstream backend_sso_cdp {
  server 127.0.0.1:9079;
}
upstream backend_arangodb {
  server 127.0.0.1:8529;
}

server {
  server_name leocdp.example.com;

  # Static Files for /view/
  location ^~ /view/ {
    # 1. Use 'alias' to map /view/ directly to the folder.
    #    Nginx automatically strips '/view/' and looks for the rest of the path inside this folder.
    alias /home/thomas/0-github/leo-cdp-framework/core-leo-cdp/resources/app-templates/leocdp-admin/;

    # 2. Check if the file exists.
    #    Note: Query parameters (?admin=1...) are ignored during file lookup, which is what we want.
    try_files $uri $uri/ =404;

    # 3. Caching and Compression settings
    add_header Cache-Control "public, max-age=2592000";
    expires 30d;

    gzip on;
    gzip_comp_level 6;
    gzip_min_length 1024;
    gzip_vary on;
    gzip_types text/css application/javascript application/x-javascript;
  }

  # Keycloak: route all /_leocdp/* to SSO backend
  location ^~ /_ssocdp/ {
    proxy_pass http://backend_admin_cdp;
    proxy_set_header Host $host;
    proxy_set_header X-Forwarded-Proto $scheme;
    proxy_set_header X-Forwarded-Port $server_port;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_read_timeout 600s;
  }

  # ArangoDB: route all /_db/* to arangodb backend
  location ^~ /_db/ {
    proxy_pass http://backend_arangodb;
    proxy_set_header Host $host;
    proxy_set_header X-Forwarded-Proto $scheme;
    proxy_set_header X-Forwarded-Port $server_port;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_read_timeout 600s;
  }

  # Proxy requests for all other paths
  location / {
    proxy_pass http://backend_admin_cdp/;
    proxy_set_header X-Forwarded-Proto $scheme;
    proxy_set_header X-Forwarded-Port $server_port;
    proxy_set_header Host $host;
    proxy_set_header X-Forwarded-For "115.74.22.116";
    access_log off;
  }

  listen [::]:443 ssl http2;
  listen 443 ssl http2;
  ssl_certificate /home/thomas/0-uspa/localhost-ssl/example.com+5.pem;
  ssl_certificate_key /home/thomas/0-uspa/localhost-ssl/example.com+5-key.pem;
}
